library(tidyverse)
read_year <- function(path) {
  name <- basename(path)
  read_csv(path) |> 
    mutate(year = parse_number(name))
}

files <- fs::dir_ls("./data/unfall/", glob = "*.csv")
years <- map(files, read_year)
full_join(years[[1]], years[[2]])
all_accidents <- reduce(years, full_join)
accidents <- years[[1]] |> 
  mutate(across(c(X,Y), ~ .x / 1e5 ))
world <- map_data("world")
world |> 
  filter(region == "Germany") |> 
  ggplot(aes(long, lat, group = subregion)) +
  geom_path() +
  coord_map()

accidents |> 
  ggplot(aes(X, Y, color = factor(ULAND))) +
  geom_point() +
  coord_map(projection = "mercator")

  count(ULAND)
